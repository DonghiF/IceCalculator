1) Sign-in
	end-point: /signIn   POST
	
	input:  nome:(Stringa)
		cognome:(Stringa)
		username(Stringa)
		password(Stringa)
		

	output: "Registrazione avvenuta con successo!"

2) Login
	end-point: /login   POST
	
	input:	username(Stringa)
		password(Stringa)
		

	output: "Benvenuto" nome + cognome 

3) Visualizza le proprie competizioni
	end-point: /Competitions   GET
	
	input:  nome:(Stringa)
		cognome:(Stringa)
		username(Stringa)
		password(Stringa)
		

	output: "Benvenuto" nome + cognome
